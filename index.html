print("欢迎来到年龄计算器！")

class User:
    def __init__(self, user_name):
        self.name = user_name

name = input("请输入您的名字：")
user1 = User(name)

while True:
    ask = input("您好，" + user1.name + "，请问需要查询您的年龄吗？(yes/no) ")

    if ask == "yes":
        tips = int(input('您今年几岁了?: '))
        print('恭喜您！您今年' + str(tips) + '岁啦！\n')

        if tips > 18:
            print("您已经不算年轻人了")
        elif tips < 18:
            print("看来您还很嫩")
        break  # 结束循环
    elif ask == "no":
        print("那下次见")
        break  # 结束循环
    else:
        print("请输入 'yes' 或 'no' 以继续。")
